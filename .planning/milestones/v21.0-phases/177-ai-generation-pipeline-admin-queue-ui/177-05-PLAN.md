    The form + card grid container. Manages local state: vertical, template_type, prompt, submitting.

    ```jsx
    import { useState } from 'react';
    import { Wand2 } from 'lucide-react';
    import { Button, Select } from '../../design-system';
    import PromptLibraryCardGrid from './PromptLibraryCardGrid';
    import { generateDraft } from '../../services/aiTemplate/templateDraftsService';

    const VERTICAL_OPTIONS = [
      { value: 'cross-vertical', label: 'Cross-vertical (any)' },
      { value: 'restaurants',    label: 'Restaurants & QSR' },
      { value: 'retail',         label: 'Retail & e-commerce' },
      { value: 'healthcare',     label: 'Healthcare & wellness' },
    ];

    const TEMPLATE_TYPE_OPTIONS = [
      { value: 'menu',         label: 'Menu' },
      { value: 'promo',        label: 'Promo' },
      { value: 'announcement', label: 'Announcement' },
      { value: 'reminder',     label: 'Reminder' },
      { value: 'wayfinding',   label: 'Wayfinding' },
      { value: 'health_tip',   label: 'Health Tip' },
    ];

    /**
     * GenerateTabForm — Phase 177 D-02 (form) + D-14 (synchronous UX with ~30s hint).
     *
     * Submit calls templateDraftsService.generateDraft (which invokes the Edge Function action=generate).
     * Edge Function does LLM + validator + retries internally and returns the draftId. UX shows a
     * blocking loader with "this can take ~30 seconds" hint — admin generates one at a time.
     */
    export default function GenerateTabForm({ onGenerated, showToast }) {
      const [vertical, setVertical] = useState('cross-vertical');
      const [templateType, setTemplateType] = useState('menu');
      const [prompt, setPrompt] = useState('');
      const [submitting, setSubmitting] = useState(false);

      const handleCardPick = (entry) => {
        setVertical(entry.vertical ?? 'cross-vertical');
        setTemplateType(entry.template_type);
        setPrompt(entry.example_freeform);
      };

      const handleSubmit = async (e) => {
        e.preventDefault();
        if (!prompt.trim()) {
          showToast?.({ variant: 'error', message: 'Prompt is required' });
          return;
        }
        setSubmitting(true);
        try {
          const verticalForApi = vertical === 'cross-vertical' ? null : vertical;
          const result = await generateDraft({
            vertical: verticalForApi,
            template_type: templateType,
            prompt: prompt.trim(),
          });
          if (result.status === 'needs_human_review') {
            showToast?.({
              variant: 'warning',
              message: `Draft created but failed validator after ${result.attempt_count} attempts — flagged for review`,
            });
          } else {
            showToast?.({
              variant: 'success',
              message: `Draft created (status: pending; attempts: ${result.attempt_count})`,
            });
          }
          onGenerated?.(result);
          setPrompt('');
        } catch (err) {
          console.error('[GenerateTabForm] generate failed:', err);
          showToast?.({ variant: 'error', message: `Generate failed: ${err.message}` });
        } finally {
          setSubmitting(false);
        }
      };

      return (
        <div className="generate-tab" data-testid="generate-tab">
          <form onSubmit={handleSubmit} className="generate-form">
            <div className="generate-form__row">
              <label htmlFor="gen-vertical">Vertical</label>
              <Select
                id="gen-vertical"
                value={vertical}
                onChange={(e) => setVertical(e.target.value)}
                disabled={submitting}
              >
                {VERTICAL_OPTIONS.map((o) => <option key={o.value} value={o.value}>{o.label}</option>)}
              </Select>
            </div>
            <div className="generate-form__row">
              <label htmlFor="gen-type">Template type</label>
              <Select
                id="gen-type"
                value={templateType}
                onChange={(e) => setTemplateType(e.target.value)}
                disabled={submitting}
              >
                {TEMPLATE_TYPE_OPTIONS.map((o) => <option key={o.value} value={o.value}>{o.label}</option>)}
              </Select>
            </div>
            <div className="generate-form__row">
              <label htmlFor="gen-prompt">Freeform prompt</label>
              <textarea
                id="gen-prompt"
                rows={4}
                value={prompt}
                onChange={(e) => setPrompt(e.target.value)}
                disabled={submitting}
                placeholder="e.g. Promote 25% off summer swimwear sale through August 31"
                data-testid="gen-prompt-textarea"
              />
            </div>
            <div className="generate-form__actions">
              <Button
                type="submit"
                variant="primary"
                disabled={submitting || !prompt.trim()}
                data-testid="gen-submit"
              >
                <Wand2 size={16} />
                {submitting ? 'Generating… (this can take ~30 seconds)' : 'Generate template'}
              </Button>
            </div>
          </form>

          <h3 className="prompt-grid__title">Or start from an example prompt</h3>
          <PromptLibraryCardGrid onPick={handleCardPick} />
        </div>
      );
    }
    ```

    **Step 2a: src/components/Admin/TemplateDraftEditModal.jsx** (~150 LOC)

    Inline modal with read-only metadata + editable svg_content + live preview + Save & Publish CTA.
    Save & Publish runs the SAME approve path the Pending-tab Approve button uses (validator → rasterize → S3 → INSERT svg_templates → mark approved); the only difference is the svg_content was edited.

    ```jsx
    import { useState, useEffect } from 'react';
    import { Save, X } from 'lucide-react';
    import { Button, Modal, Alert } from '../../design-system';
    import TemplateDraftPreview from './TemplateDraftPreview';
    import { saveDraftSvgContent, approveDraft } from '../../services/aiTemplate/templateDraftsService';
    import { validateSvg } from '../../services/svgValidator';

    /**
     * TemplateDraftEditModal — Phase 177 D-04 OVERRIDE (inline edit modal, NOT AdminEditTemplatePage extension).
     *
     * Why inline: AdminEditTemplatePage operates on template_library (Polotno templates) via marketplaceService;
     * NOT on svg_templates / template_drafts. See RESEARCH §"Existing-Code Inspection" for the verdict.
     *
     * Save & Publish runs the same approve path the Pending tab Approve button uses (validator runs
     * server-side at INSERT — defense-in-depth: we ALSO validate client-side here before invoking
     * approveDraft, so admin gets immediate feedback on hand-edited issues).
     */
    export default function TemplateDraftEditModal({ draft, onClose, onSaved, showToast }) {
      const [editedSvg, setEditedSvg] = useState(draft?.svg_content ?? '');
      const [saving, setSaving] = useState(false);
      const [validationErrors, setValidationErrors] = useState([]);
      const [validationWarnings, setValidationWarnings] = useState([]);

      useEffect(() => { setEditedSvg(draft?.svg_content ?? ''); }, [draft?.id]);

      const handleRevalidate = () => {
        const result = validateSvg(editedSvg);
        setValidationErrors(result.errors ?? []);
        setValidationWarnings(result.warnings ?? []);
        if (result.ok) showToast?.({ variant: 'success', message: 'Validator passed' });
        else            showToast?.({ variant: 'error', message: `Validator failed: ${result.errors.join('; ')}` });
      };

      const handleSaveAndPublish = async () => {
        // Client-side defense-in-depth gate (mirrors BulkTemplateUpload.jsx:174-190).
        // Server-side EF approve will re-run the validator with deno-dom — this catch is
        // only to give the admin immediate UX feedback on edited SVG.
        const result = validateSvg(editedSvg);
        if (!result.ok) {
          setValidationErrors(result.errors);
          showToast?.({ variant: 'error', message: `Cannot publish — validator failed: ${result.errors.join('; ')}` });
          return;
        }
        if ((result.warnings ?? []).length > 0) {
          console.warn('[TemplateDraftEditModal] warnings:', result.warnings);
        }

        setSaving(true);
        try {
          // Step 1 — persist the edited svg_content
          await saveDraftSvgContent(draft.id, editedSvg);
          // Step 2 — invoke EF approve (which re-validates + rasterizes + S3 + INSERTs + marks approved)
          const res = await approveDraft(draft.id);
          showToast?.({ variant: 'success', message: `Saved & published — thumbnail at ${res.thumbnail_url}` });
          await onSaved?.();
        } catch (err) {
          console.error('[TemplateDraftEditModal] save & publish failed:', err);
          showToast?.({ variant: 'error', message: `Save & publish failed: ${err.message}` });
        } finally {
          setSaving(false);
        }
      };

      if (!draft) return null;

      return (
        <Modal
          open
          onClose={() => !saving && onClose()}
          size="lg"
          aria-labelledby="edit-draft-title"
        >
          <div className="p-6 space-y-4" data-testid="edit-draft-modal">
            <div className="flex items-start justify-between">
              <h3 id="edit-draft-title" className="text-lg font-semibold">Edit draft</h3>
              <button type="button" onClick={onClose} disabled={saving} aria-label="Close" className="p-1">
                <X size={18} />
              </button>
            </div>

            {/* Read-only metadata block (D-04 override spec) */}
            <div className="meta-grid bg-gray-50 p-3 rounded text-sm">
              <div><strong>Prompt:</strong> {draft.prompt}</div>
              <div><strong>Vertical:</strong> {draft.vertical ?? 'cross-vertical'}</div>
              <div><strong>Template type:</strong> {draft.metadata?.template_type ?? '—'}</div>
              <div><strong>Attempts:</strong> {draft.metadata?.attempt_count ?? 1}</div>
            </div>

            {/* Live preview */}
            <div>
              <label className="block text-sm font-medium mb-1">Live preview</label>
              <TemplateDraftPreview
                svgContent={editedSvg}
                style={{ width: '100%', maxHeight: 300, border: '1px solid #e5e7eb', borderRadius: 4 }}
              />
            </div>

            {/* Editable SVG textarea */}
            <div>
              <label htmlFor="edit-svg" className="block text-sm font-medium mb-1">SVG content</label>
              <textarea
                id="edit-svg"
                rows={10}
                value={editedSvg}
                onChange={(e) => setEditedSvg(e.target.value)}
                disabled={saving}
                className="w-full font-mono text-xs border rounded p-2"
                data-testid="edit-svg-textarea"
              />
            </div>

            {validationErrors.length > 0 && (
              <Alert variant="error" data-testid="edit-validation-errors">
                <strong>Validator errors:</strong>
                <ul className="list-disc ml-6">
                  {validationErrors.map((e, i) => <li key={i}>{e}</li>)}
                </ul>
              </Alert>
            )}
            {validationWarnings.length > 0 && (
              <Alert variant="warning">
                <strong>Validator warnings:</strong>
                <ul className="list-disc ml-6">
                  {validationWarnings.map((w, i) => <li key={i}>{w}</li>)}
                </ul>
              </Alert>
            )}

            <div className="flex justify-end gap-2 pt-2 border-t">
              <Button variant="secondary" onClick={handleRevalidate} disabled={saving} data-testid="btn-revalidate">
                Re-validate
              </Button>
              <Button variant="secondary" onClick={onClose} disabled={saving}>Cancel</Button>
              <Button
                variant="primary"
                onClick={handleSaveAndPublish}
                disabled={saving || !editedSvg.trim()}
                data-testid="btn-save-publish"
              >
                <Save size={14} /> {saving ? 'Saving…' : 'Save & Publish'}
              </Button>
            </div>
          </div>
        </Modal>
      );
    }
    ```
