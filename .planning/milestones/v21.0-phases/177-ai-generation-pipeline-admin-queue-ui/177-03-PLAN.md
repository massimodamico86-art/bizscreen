    // Phase 177 — @resvg/resvg-wasm WASM init + rasterize helper.
    // D-15 OVERRIDE: WASM (not @resvg/resvg-js); Supabase Edge Functions only support WASM image libs.
    // index_bg.wasm is bundled via supabase/config.toml [functions.generate-svg-template] static_files.
    import { Resvg, initWasm } from "npm:@resvg/resvg-wasm@^2.6.2";

    let wasmReady: Promise<void> | null = null;

    export async function ensureWasm(): Promise<void> {
      if (!wasmReady) {
        const wasmBytes = await Deno.readFile(new URL("./index_bg.wasm", import.meta.url));
        wasmReady = initWasm(wasmBytes);
      }
      return wasmReady;
    }

    export async function rasterize(
      svg: string,
      opts: { width: number; height: number },
    ): Promise<Uint8Array> {
      await ensureWasm();
      const useWidthFit = opts.width >= opts.height;
      const resvg = new Resvg(svg, {
        fitTo: useWidthFit
          ? { mode: "width", value: opts.width }
          : { mode: "height", value: opts.height },
        background: "rgba(255,255,255,1)",
        font: { loadSystemFonts: false },     // Pitfall 2 — WASM has no system fonts; prompts force sans-serif
      });
      return resvg.render().asPng();
    }

    ```typescript
    // Phase 177 — approve handler. TADM-03 atomicity: re-validate → rasterize → S3 PUT → INSERT svg_templates → UPDATE draft.
    // Failure of any step (incl. server-side validator) leaves draft in 'pending' so admin can retry. No partial publish.
    // B1: validateSvg re-runs at approve to defend against admins editing svg_content via the Edit modal
    //     (client-side validation can be bypassed via devtools — never trust client validation alone).
    import { rasterize } from "../resvg-wasm-init.ts";
    import { uploadPng } from "../s3.ts";
    import { validateSvg } from "../svgValidator.ts";
    import { DOMParser as DenoDOMParser } from "jsr:@b-fuze/deno-dom";

    interface ApproveBody { draftId: string; reviewerUid: string; }
    interface ApproveResult { ok: true; thumbnail_url: string; svg_template_id: string; }

    // Default thumbnail dimensions — matches scripts/generate-template-thumbnails.cjs precedent.
    const THUMB_LANDSCAPE = { width: 480, height: 270 };
    const THUMB_PORTRAIT  = { width: 270, height: 480 };

    function parseSvgDimensions(svg: string): { width: number; height: number; orientation: "landscape" | "portrait" } {
      // viewBox="0 0 W H" — capture group 4-tuple
      const m = svg.match(/viewBox\s*=\s*["']\s*[\d.]+\s+[\d.]+\s+([\d.]+)\s+([\d.]+)\s*["']/);
      if (m) {
        const w = parseFloat(m[1]); const h = parseFloat(m[2]);
        if (w > 0 && h > 0) return { width: w, height: h, orientation: w >= h ? "landscape" : "portrait" };
      }
      // Fallback: standard 1920×1080 (matches prompt-library default)
      return { width: 1920, height: 1080, orientation: "landscape" };
    }

    function deriveNameFromDraft(draft: any): string {
      const tt = draft.metadata?.template_type ?? "template";
      const v  = draft.vertical ?? "general";
      const ts = new Date().toISOString().slice(0, 10);
      return `AI ${tt} (${v}) — ${ts}`;
    }
    function deriveDescriptionFromDraft(draft: any): string {
      return (draft.prompt ?? "").slice(0, 240);
    }
    function deriveCategoryFromDraft(draft: any): string {
      // Phase 178 may refine this; for v1 use the template_type as a category surrogate
      // (existing chk_svg_templates_category_enum permits these via Phase 175 taxonomy work)
      const tt = draft.metadata?.template_type ?? "general";
      const map: Record<string, string> = {
        menu: "Menu",
        promo: "Promotion",
        announcement: "Announcement",
        reminder: "Reminder",
        wayfinding: "Wayfinding",
        health_tip: "Health & Safety",
      };
      return map[tt] ?? "General";
    }

    // S3 upload moved to ./s3.ts — see Step 1c below. Approve.ts imports `uploadPng` from there.
    // (B3 fix — the legacy Vite dev presign route is dev-only middleware; no production endpoint exists.)

    export async function approve(body: ApproveBody, supabase: any): Promise<ApproveResult> {
      // Step 1 — load draft
      const { data: draft, error: fetchErr } = await supabase
        .from("template_drafts")
        .select("*")
        .eq("id", body.draftId)
        .single();
      if (fetchErr || !draft) throw new Error(`Draft not found: ${body.draftId}`);
      if (draft.status === "approved") {
        // Idempotency — return current thumbnail URL if previously approved
        return { ok: true, thumbnail_url: draft.metadata?.thumbnail_url ?? "", svg_template_id: draft.metadata?.svg_template_id ?? "" };
      }

      // Step 1b — RE-VALIDATE draft.svg_content (B1 — defense-in-depth).
      // The Edit modal can mutate svg_content; client-side validateSvg can be bypassed via devtools.
      // We re-run the validator here so no malformed/XSS SVG reaches svg_templates / the gallery.
      // DOMPurify=null per RESEARCH §"Constraint 4" (Rule 4 silently skipped server-side; the inline
      // preview re-sanitizes via DOMPurify when admins view it). DOMParserCtor=DenoDOMParser.
      const validatorResult = validateSvg(draft.svg_content, {
        DOMParserCtor: DenoDOMParser,
        DOMPurify: null,
      });
      if (!validatorResult.ok) {
        // 422 — semantically unprocessable. Draft remains 'pending' so admin can fix or reject.
        throw new Response(
          JSON.stringify({
            error: "Draft failed re-validation at approve",
            issues: validatorResult.errors,
          }),
          { status: 422, headers: { "Content-Type": "application/json" } },
        );
      }

      const dim = parseSvgDimensions(draft.svg_content);
      const thumbDim = dim.orientation === "landscape" ? THUMB_LANDSCAPE : THUMB_PORTRAIT;

      // Step 2 — rasterize (only after validator gates the content)
      const png = await rasterize(draft.svg_content, thumbDim);
      if (png.byteLength < 100) throw new Error("Rasterize produced suspiciously small PNG (<100 bytes)");

      // Step 3 — S3 PUT (B3 — direct AWS SDK upload via ./s3.ts; no presign route)
      const fileUrl = await uploadPng(png, `ai-${draft.id}`);

      // Step 4a — INSERT svg_templates
      const { data: tpl, error: insertErr } = await supabase.from("svg_templates").insert({
        name: deriveNameFromDraft(draft),
        description: deriveDescriptionFromDraft(draft),
        category: deriveCategoryFromDraft(draft),
        orientation: dim.orientation,
        thumbnail: fileUrl,
        svg_url: "",
        svg_content: draft.svg_content,
        width: Math.round(dim.width),
        height: Math.round(dim.height),
        tags: [],
        is_active: true,
        vertical: draft.vertical,
        metadata: { source: "ai_generation", draft_id: draft.id, prompt: draft.prompt, model_id: draft.metadata?.model_id ?? null },
      }).select("id").single();
      if (insertErr) throw insertErr;

      // Step 4b — UPDATE draft (only after svg_templates INSERT succeeds — preserves "no partial publish" guarantee)
      const { error: updateErr } = await supabase
        .from("template_drafts")
        .update({
          status: "approved",
          metadata: {
            ...(draft.metadata ?? {}),
            reviewed_by: body.reviewerUid,
            reviewed_at: new Date().toISOString(),
            thumbnail_url: fileUrl,
            svg_template_id: tpl.id,
          },
        })
        .eq("id", body.draftId);
      if (updateErr) throw updateErr;

      return { ok: true, thumbnail_url: fileUrl, svg_template_id: tpl.id };
    }
    ```

    interface RejectBody { draftId: string; reason?: string; reviewerUid: string; }
    interface RejectResult { ok: true; draftId: string; }

    export async function reject(body: RejectBody, supabase: any): Promise<RejectResult> {
      const { data: draft, error: fetchErr } = await supabase
        .from("template_drafts")
        .select("metadata, status")
        .eq("id", body.draftId)
        .single();
      if (fetchErr || !draft) throw new Error(`Draft not found: ${body.draftId}`);
      if (draft.status === "approved") throw new Error("Cannot reject an already-approved draft");

      const { data, error } = await supabase
        .from("template_drafts")

          status: "rejected",

        })

        .select("id")
        .single();
