**Table Partitioning:**

PostgreSQL cannot add partitioning to an existing table. Follow the rename-and-replace strategy:

```sql
-- 1. Create new partitioned table with same schema
CREATE TABLE public.playback_events_new (
  LIKE public.playback_events INCLUDING ALL
) PARTITION BY RANGE (created_at);

-- 2. Create monthly partitions for 2026 (Jan through Dec)
-- plus a few months into 2027 for headroom
CREATE TABLE public.playback_events_y2026m01
  PARTITION OF public.playback_events_new
  FOR VALUES FROM ('2026-01-01') TO ('2026-02-01');
-- ... repeat for each month through 2027m03

-- 3. Create DEFAULT partition for any unmatched data
CREATE TABLE public.playback_events_default
  PARTITION OF public.playback_events_new DEFAULT;

-- 4. Copy existing data
INSERT INTO public.playback_events_new SELECT * FROM public.playback_events;

-- 5. Swap tables
ALTER TABLE public.playback_events RENAME TO playback_events_old;
ALTER TABLE public.playback_events_new RENAME TO playback_events;

-- 6. Recreate indexes on partitioned table
-- (LIKE ... INCLUDING ALL copies constraints but indexes on partitioned table
-- need to be explicitly created)
CREATE INDEX IF NOT EXISTS idx_pe_tenant_started ON public.playback_events(tenant_id, started_at);
CREATE INDEX IF NOT EXISTS idx_pe_screen_started ON public.playback_events(screen_id, started_at);
CREATE INDEX IF NOT EXISTS idx_pe_created_at ON public.playback_events(created_at);

-- 7. Recreate RLS
ALTER TABLE public.playback_events ENABLE ROW LEVEL SECURITY;
-- Recreate existing RLS policies from original table

-- 8. Drop old table (safe -- data has been copied)
DROP TABLE public.playback_events_old;
```

Generate partitions for months 2026-01 through 2027-03 (15 months) using a DO block loop to avoid repetition:
```sql
DO $$
DECLARE
  start_date DATE := '2026-01-01';
  end_date DATE;
  partition_name TEXT;
BEGIN
  FOR i IN 0..14 LOOP
    end_date := start_date + INTERVAL '1 month';
    partition_name := 'playback_events_y' || to_char(start_date, 'YYYY') || 'm' || to_char(start_date, 'MM');
    EXECUTE format(
      'CREATE TABLE IF NOT EXISTS public.%I PARTITION OF public.playback_events_new FOR VALUES FROM (%L) TO (%L)',
      partition_name, start_date, end_date
    );
    start_date := end_date;
  END LOOP;
END $$;
```

**Auto-partition pg_cron job:**

Create a function that auto-creates the next month's partition:
```sql
CREATE OR REPLACE FUNCTION public.create_next_playback_partition()
RETURNS void
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
  next_month DATE := date_trunc('month', now()) + INTERVAL '2 months';
  partition_name TEXT := 'playback_events_y' || to_char(next_month, 'YYYY') || 'm' || to_char(next_month, 'MM');
  end_date DATE := next_month + INTERVAL '1 month';
BEGIN
  -- Check if partition already exists
  IF NOT EXISTS (
    SELECT 1 FROM pg_class WHERE relname = partition_name
  ) THEN
    EXECUTE format(
      'CREATE TABLE public.%I PARTITION OF public.playback_events FOR VALUES FROM (%L) TO (%L)',
      partition_name, next_month, end_date
    );
    RAISE NOTICE 'Created partition: %', partition_name;
  END IF;
END;
$$;
```

Schedule with pg_cron (run on 25th of each month):
```sql
SELECT cron.schedule(
  'create_monthly_playback_partition',
  '0 0 25 * *',
  $$ SELECT public.create_next_playback_partition(); $$
);
```
