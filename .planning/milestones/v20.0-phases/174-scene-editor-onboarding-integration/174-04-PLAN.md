    ```javascript
    /**
     * Phase 174 D-06: Thin client wrapper over apply_template_to_active_slide RPC.
     *
     * Overwrites the active slide's design_json.svgContent in a single DB transaction.
     * The RPC enforces:
     *   - Auth: auth.uid() must be present
     *   - Scene ownership (auth.uid() owns scene OR super_admin)
     *   - Slide belongs to the named scene
     *   - editor_type === 'svg' (rejects polotno per D-02)
     *   - Template visibility per gallery_templates VIEW RLS predicate
     *
     * @param {string} sceneId      - UUID of origin scene (returnSceneId from URL)
     * @param {string} slideId      - UUID of active slide to overwrite
     * @param {string} templateId   - UUID of svg_templates row to apply
     * @param {string} editorType   - must be 'svg' (RPC rejects others server-side)
     * @returns {Promise<string>} updated slide UUID (RPC return value)
     * @throws {Error} on any RPC error including: not authenticated, ownership
     *   denied, slide-not-in-scene, polotno rejection, template not found, no SVG body
     */
    export async function applyTemplateToActiveSlide(sceneId, slideId, templateId, editorType) {
      const { data, error } = await supabase.rpc('apply_template_to_active_slide', {
        p_scene_id:    sceneId,
        p_slide_id:    slideId,
        p_template_id: templateId,
        p_editor_type: editorType,
      });
      if (error) throw error;
      return data;
    }
    ```

    ```javascript
    const handleApply = async () => {
      if (!current) return;
      setApplying(true);
      setError(null);
      try {
        if (mode === 'editor-return') {
          // D-06: mutate existing slide in origin scene; do NOT clone-to-new-scene.
          // Polotno is filtered out at the gallery layer (D-02), and the RPC also
          // rejects non-svg editor_type as defense-in-depth.
          if (!returnSceneId || !returnSlideId) {
            throw new Error('editor-return mode requires returnSceneId and returnSlideId props');
          }
          await applyTemplateToActiveSlide(
            returnSceneId,
            returnSlideId,
            current.id,
            current.editor_type,
          );
          // D-04: navigate to scene-editor-<returnSceneId> (App.jsx:1017–1020 page-key).

        } else {

          const sceneId = await applyTemplate(current, { customizedSvg });
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client → marketplaceService | Plain JS call from React component, args are React state (already trusted at the component layer) |
| marketplaceService → supabase.rpc | RPC call crosses to DB — the RPC enforces ALL real security checks |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-174-04-01 | Tampering | applyTemplateToActiveSlide wrapper | accept | Wrapper is a thin pass-through. No client-side validation of args (consistent with applyStarterPack precedent). All security enforcement is in the DB RPC (Plan 02 mitigations T-174-02-01 through T-174-02-05). |
| T-174-04-02 | Information Disclosure | error message leakage | mitigate | catch block in handleApply uses a generic user-facing string ("Couldn't apply template...") and console.error for the raw err. RPC errors (e.g. "Scene not found or access denied") are not surfaced to the UI verbatim — prevents leaking ownership signal to attackers. |
| T-174-04-03 | Tampering | mode prop tampering by parent component | accept | The mode prop is set by TemplateGalleryPage based on the URL param it reads. An attacker can already write `?editorReturn=1` themselves; the worst case is they trigger the editor-return path with their own returnSceneId/returnSlideId — and the DB RPC's ownership check rejects any scene/slide they don't own. |

ASVS Level 1: V4 (Access Control) — defense-in-depth at the DB layer; client wrapper does not weaken it.
</threat_model>
