        await expect(page.getByRole('heading', { name: /^Templates$/ })).toBeVisible();
        // Filter bar is present
        await expect(page.getByPlaceholder('Search templates...')).toBeVisible();
        // No error toast
        await expect(page.locator('[role="alert"]')).toHaveCount(0);
        // At least one template card OR an empty-state (both are valid — TQAL-05: no exact count)
        const hasCards = await page.locator('h3').count();
        expect(hasCards).toBeGreaterThanOrEqual(0);
      });

      test('search filters instantly (TDSC-01)', async ({ page }) => {
        await navigateToSection(page, 'templates');
        const search = page.getByPlaceholder('Search templates...');
        await expect(search).toBeVisible();
        await search.fill('zzzzzz-no-match-query-xyz');
        // 150ms debounce + fuse.search + render — no-results empty state must appear within 1.5s
        await expect(page.getByText('No templates match your search')).toBeVisible({ timeout: 1500 });
        await expect(page.getByRole('button', { name: 'Browse all templates' })).toBeVisible();
      });

      test('clear all resets search (TDSC-03)', async ({ page }) => {
        await navigateToSection(page, 'templates');
        const search = page.getByPlaceholder('Search templates...');
        await search.fill('zzzzzz-no-match-query-xyz');
        await expect(page.getByText('No templates match your search')).toBeVisible({ timeout: 1500 });
        // Click "Browse all templates" (the empty-state action is the clear-all button)
        await page.getByRole('button', { name: 'Browse all templates' }).click();
        // URL must clear its query string
        await expect.poll(() => new URL(page.url()).search).toBe('');
      });

      test('URL-synced filters restore state (TDSC-04)', async ({ page }) => {
        await navigateToSection(page, 'templates');
        const base = page.url().split('?')[0];
        // Round-trip: navigate directly to ?orientation=landscape, reload, confirm state reflects
        await page.goto(`${base}?orientation=landscape&sort=alpha`);

        await expect(page.getByRole('button', { name: /^Landscape$/i })).toBeVisible();

        await expect(sortSelect).toHaveValue('alpha');
        // Active-filter chip row shows orientation
        await expect(page.getByText(/Orientation: landscape/i)).toBeVisible();
      });

      test('mobile single-column layout (TGAL-05)', async ({ page }) => {
