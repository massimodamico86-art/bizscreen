The SVG editor needs to be opened first. Create a shared setup approach:

```javascript
async function openSvgEditor(page) {
  // Navigate to Templates page via sidebar
  const templatesBtn = page.locator('aside').locator('button, a').filter({ hasText: /templates/i }).first();
  await templatesBtn.click();
  await page.waitForTimeout(2000);

  // Look for a template card and click to open editor
  // Templates page has template cards - look for "Use Template" or click on a card
  const templateCard = page.locator('article, [class*="card"], [class*="Card"]').filter({ has: page.locator('img') }).first();
  const useBtn = page.locator('button').filter({ hasText: /use template|customize|edit|open/i }).first();

  // Try clicking "New Design" or "Create" button first (creates blank canvas)
  const newDesignBtn = page.locator('button').filter({ hasText: /new design|create.*design|blank/i }).first();

  if (await newDesignBtn.isVisible({ timeout: 3000 }).catch(() => false)) {
    await newDesignBtn.click();
  } else if (await useBtn.isVisible({ timeout: 3000 }).catch(() => false)) {
    await useBtn.click();
  } else if (await templateCard.isVisible({ timeout: 3000 }).catch(() => false)) {
    await templateCard.click();
  }

  // Wait for SVG editor to load (canvas element or toolbar)
  await page.waitForTimeout(3000);
  // Look for canvas or toolbar indicators
  await page.locator('canvas, [class*="toolbar"], [class*="Toolbar"]').first().waitFor({ state: 'visible', timeout: 15000 }).catch(() => {});
}
```
