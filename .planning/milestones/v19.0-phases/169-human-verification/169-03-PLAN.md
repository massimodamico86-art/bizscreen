    **Reporter strategy (B3 fix)**: `--reporter=list,json` streams JSON to stdout and does NOT write `test-results.json` on disk. To get both human-readable output AND a per-run JSON timing artifact, use TWO separate invocations per run: one `--reporter=list` for the human-readable tail (piped through `tee` with `pipefail`/`PIPESTATUS[0]` to preserve playwright's exit code — W5), and one `--reporter=json` redirected via `PLAYWRIGHT_JSON_OUTPUT_FILE` to a per-run file we can later mine for the timing histogram.

    ```
    set -o pipefail
    mkdir -p /tmp/169-hver05

    for i in 1 2 3; do
      echo "=== Run $i (list reporter for human output) ==="
      npx playwright test tests/e2e/player-rendering.spec.js tests/e2e/player-offline-selfheal.spec.js tests/e2e/player-telemetry.spec.js --project=chromium --reporter=list --retries=0 2>&1 | tee /tmp/169-hver05/run-$i.txt
      LIST_EXIT=${PIPESTATUS[0]}
      echo "list-run $i exit=$LIST_EXIT"

      echo "=== Run $i (json reporter for timing histogram) ==="
      PLAYWRIGHT_JSON_OUTPUT_FILE=/tmp/169-hver05/run-$i.json npx playwright test tests/e2e/player-rendering.spec.js tests/e2e/player-offline-selfheal.spec.js tests/e2e/player-telemetry.spec.js --project=chromium --reporter=json --retries=0 > /tmp/169-hver05/run-$i.json.stdout 2>&1
      JSON_EXIT=$?
      echo "json-run $i exit=$JSON_EXIT"

      # Both invocations must agree on exit status. If they diverge, treat as instability (flake between back-to-back runs) and surface in evidence doc.
      if [ "$LIST_EXIT" != "$JSON_EXIT" ]; then
        echo "WARNING: run $i list-exit=$LIST_EXIT vs json-exit=$JSON_EXIT — flake between consecutive runs" | tee -a /tmp/169-hver05/flake.log
      fi
    done
    ```
