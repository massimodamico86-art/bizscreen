**Step 1: Add `installWithCustomization()` to `src/services/marketplaceService.js`**

Add after `installTemplateAsScene()` (after line 200):

```javascript
/**
 * Install a template as a new scene with customized SVG content
 * @param {string} templateId - Template UUID
 * @param {string} sceneName - Scene name
 * @param {string} customizedSvg - Modified SVG string from QuickCustomizePanel
 * @returns {Promise<string>} New scene ID
 */
export async function installWithCustomization(templateId, sceneName, customizedSvg) {
  // Step 1: Clone the template to create a new scene (reuses existing RPC)
  const sceneId = await installTemplateAsScene(templateId, sceneName);

  // Step 2: Fetch the cloned scene's first slide and patch its design_json with customized SVG
  const { data: slides, error: fetchError } = await supabase
    .from('scene_slides')
    .select('id, design_json')
    .eq('scene_id', sceneId)
    .order('position', { ascending: true })
    .limit(1);

  if (fetchError) throw fetchError;

  if (slides?.[0]) {
    const updatedDesignJson = {
      ...(slides[0].design_json || {}),
      svgContent: customizedSvg,
    };
    const { error: updateError } = await supabase
      .from('scene_slides')
      .update({ design_json: updatedDesignJson })
      .eq('id', slides[0].id);

    if (updateError) throw updateError;
  }

  return sceneId;
}
```
