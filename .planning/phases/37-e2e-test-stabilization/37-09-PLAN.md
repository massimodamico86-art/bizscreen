         networkErrors.push({

         });

     await page.waitForLoadState('domcontentloaded');

     const dialog = page.locator('[role="dialog"]');
     if (await dialog.count() > 0 && await dialog.isVisible()) {

       await dialog.waitFor({ state: 'hidden', timeout: 2000 }).catch(() => {});
