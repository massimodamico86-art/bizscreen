- On click: scan canvas objects and find the most common fill color, replace it with the clicked brand color. Use this approach:
  ```
  const handleApplyColor = (newColor) => {
    if (!canvas) return;
    // Collect all fill colors and find the most common non-white, non-black
    const colorCounts = {};
    canvas.getObjects().forEach(obj => {
      const fill = obj.fill;
      if (fill && typeof fill === 'string' && fill !== '#ffffff' && fill !== '#FFFFFF' && fill !== '#000000') {
        colorCounts[fill] = (colorCounts[fill] || 0) + 1;
      }
    });
    const dominantColor = Object.entries(colorCounts).sort((a, b) => b[1] - a[1])[0]?.[0];
    if (!dominantColor) return;

    canvas.getObjects().forEach(obj => {
      if (obj.fill === dominantColor) obj.set('fill', newColor);
      if (obj.stroke === dominantColor) obj.set('stroke', newColor);
    });
    canvas.renderAll();
    onCanvasModified?.();
  };
  ```

  const handlePlaceLogo = async () => {
    const logoUrl = brandTheme?.logo_url;
    if (!logoUrl) return;
    try {
      const img = await fabric.FabricImage.fromURL(logoUrl, { crossOrigin: 'anonymous' });
      const maxWidth = canvas.width * 0.2;
      const maxHeight = canvas.height * 0.15;
      const scale = Math.min(maxWidth / img.width, maxHeight / img.height, 1);
      img.set({

        scaleX: scale,
        scaleY: scale,
        id: `logo-${Date.now()}`,
        name: 'Brand Logo',

      canvas.add(img);
      canvas.setActiveObject(img);
      canvas.renderAll();
      onCanvasModified?.();

    }
  };
