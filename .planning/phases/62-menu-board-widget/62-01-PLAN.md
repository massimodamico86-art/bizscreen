    **Currency Formatting (MENU-09):**
    ```javascript
    export function formatMenuPrice(amount, currencyCode, locale) {
      if (amount === null || amount === undefined || amount === '') return '';
      const num = Number(amount);
      if (isNaN(num)) return '';
      try {
        return new Intl.NumberFormat(locale || 'en-US', {
          style: 'currency',
          currency: currencyCode || 'USD',
        }).format(num);
      } catch {
        return `$${num.toFixed(2)}`;
      }
    }
    ```

    **Realtime Subscription (MENU-07):**
    ```javascript
    export function subscribeToMenuBoard(menuBoardId, onUpdate) {
      const channel = supabase
        .channel(`menu-board:${menuBoardId}`)
        .on('postgres_changes', {
          event: '*',
          schema: 'public',
          table: 'menu_items',
          filter: `menu_board_id=eq.${menuBoardId}`,
        }, (payload) => {
          onUpdate({ table: 'menu_items', ...payload });
        })
        .on('postgres_changes', {
          event: '*',
          schema: 'public',
          table: 'menu_categories',
          filter: `menu_board_id=eq.${menuBoardId}`,
        }, (payload) => {
          onUpdate({ table: 'menu_categories', ...payload });
        })
        .subscribe();

      return {
        unsubscribe: async () => {
          await supabase.removeChannel(channel);
        },
      };
    }
    ```
    Use a SINGLE channel with two `.on()` calls (not two separate channels) to reduce connection overhead. This follows the pattern from realtimeService.js.
