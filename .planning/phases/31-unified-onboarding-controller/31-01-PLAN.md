  <action>
Create a custom React hook that wraps the Phase 30 onboarding API.

The hook should:
1. Load initial state on mount using `getUnifiedOnboardingState()`
2. Expose `state` object with: currentStep, canResume, progressPercent, isComplete, skippedAt
3. Expose `loading` boolean (true while initial fetch in progress)
4. Expose `error` state (null or Error object)
5. Expose `advance(step)` function that calls `advanceOnboardingStep(step)` and refreshes state
6. Expose `skip()` function that calls `skipOnboarding()` and refreshes state
7. Expose `refresh()` function to manually re-fetch state
8. Add visibility change listener to refresh when user returns to tab (handles multi-tab sync)

Pattern to follow from RESEARCH.md:
```javascript
import { useState, useEffect, useCallback } from 'react';
import { getUnifiedOnboardingState, advanceOnboardingStep, skipOnboarding } from '../services/onboardingService';

export function useUnifiedOnboarding() {
  const [state, setState] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  const refresh = useCallback(async () => {
    try {
      const data = await getUnifiedOnboardingState();
      setState(data);
      setError(null);
    } catch (err) {
      setError(err);
    }
  }, []);

  // Initial load
  useEffect(() => {
    async function init() {
      setLoading(true);
      await refresh();
      setLoading(false);
    }
    init();
  }, [refresh]);

  // Visibility change listener
  useEffect(() => {
    const handleVisibilityChange = () => {
      if (document.visibilityState === 'visible') {
        refresh();
      }
    };
    document.addEventListener('visibilitychange', handleVisibilityChange);
    return () => document.removeEventListener('visibilitychange', handleVisibilityChange);
  }, [refresh]);

  const advance = useCallback(async (step) => {
    const result = await advanceOnboardingStep(step);
    if (result.success) {
      await refresh();
    }
    return result;
  }, [refresh]);

  const skip = useCallback(async () => {
    const result = await skipOnboarding();
    if (result.success) {
      await refresh();
    }
    return result;
  }, [refresh]);

  return { state, loading, error, advance, skip, refresh };
}
```

- Visual only - no percentage text or step count displayed
- Show skeleton during loading to avoid 0% flash

Implementation:
```jsx
import { motion } from 'framer-motion';
import PropTypes from 'prop-types';

/**
 * Thin progress bar showing onboarding completion percentage
 * @param {Object} props
 * @param {number} props.progress - Percentage complete (0-100)
 * @param {boolean} [props.loading] - Show skeleton state
 */
export function OnboardingProgressBar({ progress, loading = false }) {
  if (loading) {
    // Skeleton state - subtle pulse animation
    return (
      <div className="h-1 bg-gray-200 rounded-full overflow-hidden">
        <div className="h-full w-1/3 bg-gray-300 animate-pulse" />
      </div>
    );
  }

  return (
    <div className="h-1 bg-gray-200 rounded-full overflow-hidden">
      <motion.div
        className="h-full bg-blue-500"
        initial={{ width: 0 }}
        animate={{ width: `${progress}%` }}
        transition={{ duration: 0.3, ease: 'easeOut' }}
      />
    </div>
  );
}

OnboardingProgressBar.propTypes = {
  progress: PropTypes.number.isRequired,
  loading: PropTypes.bool,
};

export default OnboardingProgressBar;
```
