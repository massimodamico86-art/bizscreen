   ```javascript
   import confetti from 'canvas-confetti';

   function triggerConfetti() {
     // Respect reduced motion preference (built-in option)
     confetti({
       particleCount: 100,
       spread: 70,
       origin: { y: 0.6 },
       disableForReducedMotion: true,
       zIndex: 10001  // Above modal overlay
     });
   }
   ```
