Add `handleGenerateFromPrompt` which finds the best matching preset by keyword matching against promptText, then applies it (this provides the "generate from prompt" capability using the existing preset library without requiring the Anthropic API in the scene editor):

```js
function handleGenerateFromPrompt() {
  if (!promptText.trim() || loading) return;
  const lowerPrompt = promptText.toLowerCase();
  // Find preset whose title or description matches prompt keywords
  const matched = presets.find(p =>
    p.title?.toLowerCase().split(' ').some(w => lowerPrompt.includes(w)) ||
    p.description?.toLowerCase().split(' ').some(w => lowerPrompt.includes(w))
  ) || presets[0]; // Fall back to first preset if no match
  if (matched) {
    handleApplyPreset(matched);
    setPromptText(''); // Clear after apply for "refinement" signal
  }
}
```

This satisfies SCEN-04: user enters a text prompt → layout is generated → they can enter a new prompt to refine (iterative).

In FabricSvgEditor, the add-image-from-URL handler pattern:
```js
const handleAddImageFromUrl = useCallback(async (url) => {
  const canvas = fabricCanvasRef.current;
  if (!canvas || !url) return;
  try {
    const img = await fabric.Image.fromURL(url, { crossOrigin: 'anonymous' });
    img.set({
      left: 100,
      top: 100,
      scaleX: Math.min(1, 400 / img.width),
      scaleY: Math.min(1, 400 / img.width),
    });
    canvas.add(img);
    canvas.setActiveObject(img);
    canvas.renderAll();
    syncCanvasObjects();
    saveHistory();
  } catch (err) {
    logger.error('Failed to add image from URL', { error: err });
    showToast?.('Failed to load cloud image', 'error');
  }
}, [syncCanvasObjects, saveHistory, showToast]);
```
